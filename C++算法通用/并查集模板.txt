const int MAXN=1e5+50;
int fa[MAXN], siz[MAXN][26];
int getFather(int x){
    return fa[x]=(fa[x]==x?x:getFather(fa[x]));
}
inline void mergeFather(int x, int y){
    int fx=getFather(x), fy=getFather(y);
    if (fx==fy) return;
    if (fx>fy) swap(fx, fy);
    fa[fx]=fy;
    for (int i=0; i<26; i++){
        siz[fy][i]+=siz[fx][i];
        siz[fx][i]=0;
    }
}
